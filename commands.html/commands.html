# order_item.rb
class OrderItem < ApplicationRecord
  belongs_to :movie
  belongs_to :cart
  belongs_to :order
end

# cart.rb
class Cart < ApplicationRecord
  belongs_to :user
  has_many :order_itens
end

# order.rb
class Order < ApplicationRecord
  belongs_to :user
  has_many :order_items
end

class User < ApplicationRecord
